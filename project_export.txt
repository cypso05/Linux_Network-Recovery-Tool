#!/bin/bash
check_bridge() {
    local bridge="${1:-$BRIDGE_IFACE}"
    local iface="${2:-$DEFAULT_IFACE}"
    local all_pass=true
    
    if [[ -z "$bridge" ]]; then
        echo "  ℹ️  No bridge configured"
        return 0
    fi
    
    if ip link show "$bridge" 2>/dev/null | grep -q "UP"; then
        echo "  ✅ Bridge $bridge is UP"
    else
        echo "  ❌ Bridge $bridge is DOWN"
        all_pass=false
    fi
    
    if bridge link show "$bridge" 2>/dev/null | grep -q "$iface"; then
        echo "  ✅ $iface enslaved to $bridge"
    else
        echo "  ❌ $iface NOT enslaved to $bridge"
        all_pass=false
    fi
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

#!/bin/bash
# diagnostics/https - HTTPS connectivity diagnostic with HTTP fallback
# Usage: check_https [url]

check_https() {
    local all_pass=true
    local https_urls=(
        "https://cloudflare.com"
        "https://google.com"
        "https://github.com"
    )
    local portal_url="http://connectivitycheck.gstatic.com/generate_204"
    
    # Allow override via argument
    if [[ -n "${1:-}" ]]; then
        https_urls=("$1")
    fi
    
    if ! command -v curl &>/dev/null; then
        echo "  ⚠️  curl not installed - HTTPS tests skipped"
        return 1
    fi
    
    echo "  📌 Testing HTTPS connectivity..."
    
    local success_count=0
    local total_urls=${#https_urls[@]}
    
    # Test HTTPS targets
    for url in "${https_urls[@]}"; do
        local http_code=$(curl -s -o /dev/null -w "%{http_code}" \
            --connect-timeout 5 --max-time 10 "$url" 2>/dev/null)
        
        if [[ "$http_code" =~ ^[23][0-9][0-9]$ ]]; then
            echo "  ✅ HTTPS: $url (HTTP $http_code)"
            ((success_count++))
        elif [[ "$http_code" == "000" ]]; then
            echo "  ❌ HTTPS: $url (connection failed)"
            all_pass=false
        else
            echo "  ⚠️  HTTPS: $url (HTTP $http_code)"
            all_pass=false
        fi
    done
    
    # Always test for captive portal (can intercept even when some HTTPS works)
    echo ""
    echo "  📌 Testing for captive portal..."
    local portal_code=$(curl -s -o /dev/null -w "%{http_code}" \
        --connect-timeout 5 --max-time 10 "$portal_url" 2>/dev/null)
    
    if [[ "$portal_code" == "204" ]]; then
        echo "  ✅ No captive portal detected"
    elif [[ "$portal_code" =~ ^[23][0-9][0-9]$ ]]; then
        echo "  ⚠️  Possible captive portal (HTTP $portal_code)"
        echo "  ℹ️  You may need to log in to the network"
        all_pass=false
    elif [[ "$portal_code" == "000" ]]; then
        echo "  ❌ Captive portal test: connection failed"
        all_pass=false
    else
        echo "  ⚠️  Captive portal test: unexpected response (HTTP $portal_code)"
        all_pass=false
    fi
    
    # Summary
    if [[ $success_count -ge 1 ]]; then
        echo "  ✅ HTTPS connectivity OK ($success_count/$total_urls)"
        return 0
    else
        echo "  ❌ HTTPS connectivity FAILED (0/$total_urls)"
        return 1
    fi
}

#!/bin/bash
# diagnostics/interface - interface layer diagnostic
# Usage: check_interface [iface]
# Enhanced with detailed interface information and production-ready output

check_interface() {
    local iface="${1:-$DEFAULT_IFACE}"
    local all_pass=true
    
    # Check if interface exists
    if [[ ! -d "/sys/class/net/$iface" ]]; then
        echo "  ❌ Interface $iface does not exist"
        return 1
    fi
    
    # Gather interface details
    local operstate=$(cat "/sys/class/net/$iface/operstate" 2>/dev/null || echo "unknown")
    local mac=$(cat "/sys/class/net/$iface/address" 2>/dev/null || echo "unknown")
    local mtu=$(cat "/sys/class/net/$iface/mtu" 2>/dev/null || echo "unknown")
    local carrier=$(cat "/sys/class/net/$iface/carrier" 2>/dev/null || echo "0")
    local speed=$(cat "/sys/class/net/$iface/speed" 2>/dev/null || echo "unknown")
    local duplex=$(cat "/sys/class/net/$iface/duplex" 2>/dev/null || echo "unknown")
    local flags=$(ip link show "$iface" 2>/dev/null | grep -oP 'flags=\K[^ ]+' || echo "unknown")
    
    # Determine interface type
    local iface_type="unknown"
    if [[ -d "/sys/class/net/$iface/wireless" ]] || [[ -d "/sys/class/net/$iface/phy80211" ]]; then
        iface_type="wireless"
    elif [[ -d "/sys/class/net/$iface/bridge" ]]; then
        iface_type="bridge"
    elif [[ -d "/sys/class/net/$iface/bonding" ]]; then
        iface_type="bond"
    elif [[ "$iface" =~ ^(vnet|virbr) ]]; then
        iface_type="virtual"
    else
        iface_type="ethernet"
    fi
    
    # Display interface info with nice formatting
    echo "  ┌─ Interface: $iface"
    echo "  │  Type: $iface_type"
    echo "  │  MAC: $mac"
    echo "  │  MTU: $mtu"
    echo "  │  Flags: $flags"
    echo "  │  State: $operstate"
    
    # Additional details based on interface type
    case "$iface_type" in
        wireless)
            local ssid=$(iwconfig "$iface" 2>/dev/null | grep -oP 'ESSID:"\K[^"]+' || echo "not connected")
            local signal=$(iwconfig "$iface" 2>/dev/null | grep -oP 'Signal level=\K[0-9-]+' || echo "unknown")
            local freq=$(iwconfig "$iface" 2>/dev/null | grep -oP 'Frequency:\K[0-9.]+ GHz' || echo "unknown")
            echo "  │  SSID: $ssid"
            echo "  │  Signal: ${signal} dBm"
            echo "  │  Frequency: $freq"
            ;;
        bridge)
            local slaves=$(bridge link show "$iface" 2>/dev/null | grep -oP '^[0-9]+: \K[^:]+' | tr '\n' ', ' | sed 's/, $//')
            echo "  │  Slaves: ${slaves:-none}"
            ;;
        ethernet)
            [[ "$speed" != "unknown" ]] && echo "  │  Speed: $speed Mb/s"
            [[ "$duplex" != "unknown" ]] && echo "  │  Duplex: $duplex"
            ;;
    esac
    
    echo "  └─────────────────────────────"
    
    # Check if carrier is present (link detected)
    if [[ "$carrier" == "1" ]]; then
        echo "  ✅ Carrier: present"
    else
        if [[ "$iface_type" == "wireless" ]]; then
            echo "  ⚠️  Carrier: no WiFi connection"
        else
            echo "  ❌ Carrier: LOST - Check cable"
        fi
        all_pass=false
    fi
    
    # Check interface state
    if [[ "$operstate" == "up" ]]; then
        echo "  ✅ Interface state: $operstate"
    elif [[ "$operstate" == "dormant" ]]; then
        echo "  ⚠️  Interface state: $operstate (waiting for carrier)"
        all_pass=false
    else
        echo "  ❌ Interface state: $operstate (should be up)"
        all_pass=false
    fi
    
    # Check if interface is UP from ip command
    if ip link show "$iface" 2>/dev/null | grep -q "UP"; then
        echo "  ✅ Interface flags: UP"
    else
        echo "  ❌ Interface flags: DOWN"
        all_pass=false
    fi
    
    # Check MTU sanity
    if [[ "$mtu" != "unknown" ]] && [[ "$mtu" -lt 1280 ]]; then
        echo "  ⚠️  MTU is very low ($mtu) - may cause issues"
        all_pass=false
    fi
    
    # Check for interface errors
    local rx_errors=$(cat "/sys/class/net/$iface/statistics/rx_errors" 2>/dev/null || echo "0")
    local tx_errors=$(cat "/sys/class/net/$iface/statistics/tx_errors" 2>/dev/null || echo "0")
    if [[ "$rx_errors" != "0" ]] || [[ "$tx_errors" != "0" ]]; then
        echo "  ⚠️  Errors: RX=$rx_errors, TX=$tx_errors"
        all_pass=false
    fi
    
    # Check for dropped packets
    local rx_dropped=$(cat "/sys/class/net/$iface/statistics/rx_dropped" 2>/dev/null || echo "0")
    local tx_dropped=$(cat "/sys/class/net/$iface/statistics/tx_dropped" 2>/dev/null || echo "0")
    if [[ "$rx_dropped" != "0" ]] || [[ "$tx_dropped" != "0" ]]; then
        echo "  ⚠️  Dropped: RX=$rx_dropped, TX=$tx_dropped"
        all_pass=false
    fi
    
    # Check for collisions
    local collisions=$(cat "/sys/class/net/$iface/statistics/collisions" 2>/dev/null || echo "0")
    if [[ "$collisions" != "0" ]]; then
        echo "  ⚠️  Collisions: $collisions"
        all_pass=false
    fi
    
    echo ""
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

#!/bin/bash
# diagnostics/physical - physical layer diagnostic
# Usage: check_physical [iface]
# Enhanced with detailed physical layer information

check_physical() {
    local all_pass=true
    local iface="${1:-$DEFAULT_IFACE}"
    
    echo "  ┌─ Physical Layer: $iface"
    
    # Check if interface exists
    if [[ ! -d "/sys/class/net/$iface" ]]; then
        echo "  │  ❌ Interface does not exist"
        echo "  └─────────────────────────────"
        return 1
    fi
    
    # Determine interface type
    local iface_type="unknown"
    if [[ -d "/sys/class/net/$iface/wireless" ]] || [[ -d "/sys/class/net/$iface/phy80211" ]]; then
        iface_type="wireless"
    elif [[ -d "/sys/class/net/$iface/bridge" ]]; then
        iface_type="bridge"
    else
        iface_type="ethernet"
    fi
    echo "  │  Type: $iface_type"
    
    # Check carrier (link detection)
    if [[ -f "/sys/class/net/$iface/carrier" ]]; then
        local carrier=$(cat "/sys/class/net/$iface/carrier")
        if [[ "$carrier" == "1" ]]; then
            echo "  │  ✅ Carrier: present"
        else
            if [[ "$iface_type" == "wireless" ]]; then
                echo "  │  ⚠️  Carrier: no WiFi connection"
            else
                echo "  │  ❌ Carrier: LOST - Check cable"
            fi
            all_pass=false
        fi
    fi
    
    # Check operstate
    if [[ -f "/sys/class/net/$iface/operstate" ]]; then
        local state=$(cat "/sys/class/net/$iface/operstate")
        case "$state" in
            up)
                echo "  │  ✅ State: $state"
                ;;
            down)
                echo "  │  ❌ State: $state (interface is down)"
                all_pass=false
                ;;
            dormant)
                echo "  │  ⚠️  State: $state (waiting for carrier)"
                all_pass=false
                ;;
            *)
                echo "  │  ❌ State: $state (unexpected)"
                all_pass=false
                ;;
        esac
    fi
    
    # Check link speed (ethernet only)
    if [[ "$iface_type" == "ethernet" ]] && command -v ethtool &>/dev/null; then
        if [[ -e "/sys/class/net/$iface/device" ]]; then
            local speed=$(timeout 2 ethtool "$iface" 2>/dev/null | grep -i "Speed:" | awk '{print $2}')
            local duplex=$(timeout 2 ethtool "$iface" 2>/dev/null | grep -i "Duplex:" | awk '{print $2}')
            local link=$(timeout 2 ethtool "$iface" 2>/dev/null | grep -i "Link detected:" | awk '{print $3}')
            
            if [[ -n "$speed" ]] && [[ "$speed" != "Unknown!" ]]; then
                echo "  │  Speed: $speed"
            fi
            if [[ -n "$duplex" ]] && [[ "$duplex" != "Unknown!" ]]; then
                echo "  │  Duplex: $duplex"
            fi
            if [[ -n "$link" ]]; then
                if [[ "$link" == "yes" ]]; then
                    echo "  │  ✅ Link detected: yes"
                else
                    echo "  │  ❌ Link detected: no"
                    all_pass=false
                fi
            fi
        fi
    fi
    
    # WiFi signal strength
    if [[ "$iface_type" == "wireless" ]] && command -v iwconfig &>/dev/null; then
        local signal=$(iwconfig "$iface" 2>/dev/null | grep -oP 'Signal level=\K[0-9-]+' || echo "")
        local quality=$(iwconfig "$iface" 2>/dev/null | grep -oP 'Quality=\K[0-9/]+' || echo "")
        if [[ -n "$signal" ]]; then
            echo "  │  Signal: ${signal} dBm"
            # Interpret signal strength
            if [[ "$signal" -gt -50 ]]; then
                echo "  │  📶 Excellent"
            elif [[ "$signal" -gt -60 ]]; then
                echo "  │  📶 Good"
            elif [[ "$signal" -gt -70 ]]; then
                echo "  │  📶 Fair"
            else
                echo "  │  📶 Poor"
            fi
        fi
        if [[ -n "$quality" ]]; then
            echo "  │  Quality: $quality"
        fi
    fi
    
    # Check for interface errors
    local rx_errors=$(cat "/sys/class/net/$iface/statistics/rx_errors" 2>/dev/null || echo "0")
    local tx_errors=$(cat "/sys/class/net/$iface/statistics/tx_errors" 2>/dev/null || echo "0")
    if [[ "$rx_errors" != "0" ]] || [[ "$tx_errors" != "0" ]]; then
        echo "  │  ⚠️  Errors: RX=$rx_errors, TX=$tx_errors"
        all_pass=false
    fi
    
    echo "  └─────────────────────────────"
    echo ""
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

#!/bin/bash
#===============================================================================
# XFCE Panel Integration for Network Recovery Tool
# Generic version - works on any XFCE panel configuration
#===============================================================================

echo "🔧 Setting up Network Recovery panel integration..."

# Check if XFCE panel is running
if ! pgrep -x "xfce4-panel" > /dev/null; then
    echo "⚠️  XFCE panel not running - integration will apply on next login"
fi

# Ensure desktop file exists
if [[ ! -f /usr/share/applications/network-recover.desktop ]]; then
    echo "❌ Desktop file not found. Run the installer first."
    exit 1
fi

# Copy to user's local applications
mkdir -p ~/.local/share/applications
cp /usr/share/applications/network-recover.desktop ~/.local/share/applications/

# Try to add launcher to the panel
if command -v xfconf-query &>/dev/null; then
    echo "📌 Adding launcher to XFCE panel..."
    
    # Detect ALL panels
    PANELS=$(xfconf-query -c xfce4-panel -p /panels -l 2>/dev/null | grep -E '^/panels/panel-[0-9]+$' | sed 's|/panels/||')
    
    if [[ -z "$PANELS" ]]; then
        echo "⚠️  No XFCE panels detected"
    else
        # Find which panel has the systray (network icon)
        TARGET_PANEL=""
        SYSTRAY_PLUGIN=""
        SYSTRAY_POS=""
        PLUGIN_LIST=""
        
        for panel in $PANELS; do
            PLUGIN_LIST=$(xfconf-query -c xfce4-panel -p /panels/$panel/plugin-ids 2>/dev/null | tr -d '[]' | tr -d ' ')
            
            if [[ -n "$PLUGIN_LIST" ]]; then
                IFS=',' read -ra PLUGINS <<< "$PLUGIN_LIST"
                
                # Check if any plugin in this panel is a systray or statusnotifier
                for plugin_id in "${PLUGINS[@]}"; do
                    PLUGIN_TYPE=$(xfconf-query -c xfce4-panel -p /plugins/plugin-$plugin_id/type 2>/dev/null)
                    
                    if [[ "$PLUGIN_TYPE" == "systray" ]] || [[ "$PLUGIN_TYPE" == "statusnotifier" ]]; then
                        TARGET_PANEL="$panel"
                        SYSTRAY_PLUGIN="$plugin_id"
                        # Find position
                        for i in "${!PLUGINS[@]}"; do
                            if [[ "${PLUGINS[$i]}" == "$plugin_id" ]]; then
                                SYSTRAY_POS=$i
                                break
                            fi
                        done
                        break 2
                    fi
                done
            fi
        done
        
        if [[ -z "$TARGET_PANEL" ]]; then
            echo "⚠️  No systray/statusnotifier plugin found (network icon not detected)"
            echo "   The launcher will be added to the first panel instead."
            
            # Fallback: use first panel
            TARGET_PANEL=$(echo "$PANELS" | head -1)
            PLUGIN_LIST=$(xfconf-query -c xfce4-panel -p /panels/$TARGET_PANEL/plugin-ids 2>/dev/null | tr -d '[]' | tr -d ' ')
            SYSTRAY_POS=-1  # Add to end
        fi
        
        # Find next available plugin ID
        if [[ -n "$PLUGIN_LIST" ]]; then
            IFS=',' read -ra PLUGINS <<< "$PLUGIN_LIST"
            NEXT_ID=$(printf "%s\n" "${PLUGINS[@]}" | sort -n | tail -1)
            NEXT_ID=$((NEXT_ID + 1))
        else
            NEXT_ID=1
        fi
        
        # Create new launcher plugin
        xfconf-query -c xfce4-panel -p /plugins/plugin-$NEXT_ID -n -t string -s "launcher" 2>/dev/null || true
        xfconf-query -c xfce4-panel -p /plugins/plugin-$NEXT_ID/items -n -t string -s "network-recover.desktop" 2>/dev/null || true
        
        # Build new plugin list with launcher inserted after systray
        if [[ -n "$PLUGIN_LIST" ]]; then
            IFS=',' read -ra PLUGINS <<< "$PLUGIN_LIST"
            NEW_LIST=""
            
            if [[ "$SYSTRAY_POS" -ge 0 ]]; then
                # Insert after systray
                for i in "${!PLUGINS[@]}"; do
                    NEW_LIST="${NEW_LIST}${PLUGINS[$i]},"
                    if [[ $i -eq $SYSTRAY_POS ]]; then
                        NEW_LIST="${NEW_LIST}${NEXT_ID},"
                    fi
                done
            else
                # Add to end
                for i in "${!PLUGINS[@]}"; do
                    NEW_LIST="${NEW_LIST}${PLUGINS[$i]},"
                done
                NEW_LIST="${NEW_LIST}${NEXT_ID},"
            fi
            
            # Remove trailing comma and wrap in brackets
            NEW_LIST="[${NEW_LIST%,}]"
            
            # Apply new plugin order
            xfconf-query -c xfce4-panel -p /panels/$TARGET_PANEL/plugin-ids -s "$NEW_LIST" 2>/dev/null || true
            
            if [[ "$SYSTRAY_POS" -ge 0 ]]; then
                echo "✅ Launcher added to panel $TARGET_PANEL (plugin $NEXT_ID) next to network icon"
            else
                echo "✅ Launcher added to panel $TARGET_PANEL (plugin $NEXT_ID)"
            fi
        else
            echo "⚠️  Could not determine plugin list"
        fi
    fi
    
    # If automated addition failed, show manual instructions
    if [[ -z "$TARGET_PANEL" ]] || [[ -z "$NEXT_ID" ]]; then
        echo ""
        echo "   To add manually:"
        echo "   1. Right-click panel → Panel → Panel Preferences"
        echo "   2. Click 'Items' tab"
        echo "   3. Click '+' Add"
        echo "   4. Search for 'Network Diagnose & Repair'"
        echo "   5. Click 'Add'"
        echo "   6. Drag it next to the network icon (usually in Status Tray)"
        echo ""
    fi
else
    echo "⚠️  xfconf-query not found - cannot auto-configure panel"
    echo ""
    echo "   Please add the launcher manually:"
    echo "   1. Right-click panel → Panel → Panel Preferences"
    echo "   2. Click 'Items' tab"
    echo "   3. Click '+' Add"
    echo "   4. Search for 'Network Diagnose & Repair'"
    echo "   5. Click 'Add'"

# Linux Network Recovery Tool

[![License: MIT](https://img.shields.io/badge/License-MIT-yellow.svg)](https://opensource.org/licenses/MIT)
[![Version](https://img.shields.io/badge/version-1.1.0-blue)](https://github.com/cypso05/Linux_Network-Recovery-Tool)
[![Platform](https://img.shields.io/badge/platform-Linux-brightgreen)](https://www.linux.org)
[![XFCE](https://img.shields.io/badge/desktop-XFCE-orange)](https://www.xfce.org)

**Network Recovery Tool** is the Linux equivalent of the Windows "Diagnose and Repair Network Problems" feature.  
It is a production‑grade, intelligent diagnostics and recovery framework that automatically detects the root cause of network failures and performs targeted repairs—**without requiring a reboot**.

---

## The Problem

We've all been there. Your network icon shows "Connected", you have a valid IP address, but:

- Web pages never load.
- DNS resolution fails.
- `ping 8.8.8.8` works, but `ping google.com` doesn't.
- The only reliable "fix" is a full system reboot.

**This is not acceptable for modern systems.**

On Windows, you right‑click the network icon and select **"Diagnose and Repair"**.  
On Linux, you're left with:

```bash
sudo systemctl restart NetworkManager   # sometimes works
sudo dhclient -v eth0                   # rarely fixes it
# …or a full reboot
Network Recovery Tool fills that gap. It provides a structured, layered approach to diagnosing and fixing network issues on any Linux distribution—desktop, server, virtualised, or containerised.

Real-World Issues This Tool Solves
Category	Specific Issues
Connectivity	"Connected but cannot browse", missing default routes, gateway unreachable, ARP cache corruption, DHCP lease failures, NetworkManager false positives
DNS	Broken resolvers, stale cache, corrupted /etc/resolv.conf, systemd-resolved failures
NetworkManager	Interface stuck in transitional state, incorrect routing updates, failed reconnection attempts
Virtualisation (KVM/Libvirt)	Bridge interfaces missing or down, slave detachment, VM networking failures
Kubernetes	K3s node‑to‑node communication failures, cluster DNS issues, host networking impacting workloads
These are not theoretical – they come from real production environments, including MX Linux desktops with KVM VMs and K3s clusters.

Vision & Core Philosophy
Network Recovery Tool aims to become the default, trusted network troubleshooter for Linux.
It is built on three key principles:

Diagnose Before Repair – Always identify the failing layer first. No blind restarts, no guessing.

Intelligent, Targeted Recovery – Apply precise repairs to the failing component (e.g., flush DNS cache, restore default route, restart only the resolver).

Evidence‑Driven – Every incident generates a timestamped report, helping you identify and fix recurring problems permanently.

Layer‑Based Diagnostics
The diagnostic process follows a strict sequential order, from the physical layer up to the application layer. The first failing layer is identified as the root cause candidate.

Physical Link – Carrier, operstate, speed/duplex

Interface State – Interface UP, flags

Bridge Health – Bridge existence, UP status, slave attachment

IP Addressing – IPv4/IPv6 presence, duplicate IP detection

Routing – Default route, multiple routes, blackholes

Gateway Reachability – ARP entry, ICMP reachability

External IP Reachability – Ping public IPs (ICMP + TCP fallback)

DNS Resolution – resolv.conf, resolution tests, poisoning detection

HTTPS Connectivity – Actual browsing simulation, captive portal, MTU

NetworkManager Health – Running state, device status, stale connections

Virtualisation Dependencies – libvirtd, VM states, vnet interfaces

Kubernetes – Cluster connectivity, node readiness (optional)

#!/bin/bash
# repairs/bridge - bridge repair action with slave re-attachment
# Usage: repair_bridge [bridge] [slave_iface]

repair_bridge() {
    local bridge="${1:-$BRIDGE_IFACE}"
    local slave="${2:-$DEFAULT_IFACE}"
    
    echo "🔧 Repairing bridge $bridge..."
    
    # Check if bridge is configured
    if [[ -z "$bridge" ]]; then
        echo "  ℹ️  No bridge configured"
        return 0
    fi
    
    # Check if bridge exists
    if ! ip link show "$bridge" &>/dev/null; then
        echo "  ❌ Bridge $bridge does not exist"
        return 1
    fi
    
    # Show current slaves
    local current_slaves=$(bridge link show "$bridge" 2>/dev/null | grep -oP '^[0-9]+: \K[^:]+' | tr '\n' ' ')
    echo "  ℹ️  Current slaves: ${current_slaves:-none}"
    
    # Check if bridge is UP
    if ip link show "$bridge" | grep -q "UP"; then
        echo "  ✅ Bridge $bridge is already UP"
    else
        echo "  📌 Bringing bridge up..."
        # Try nmcli first
        if command -v nmcli &>/dev/null && nmcli con up "$bridge" 2>/dev/null; then
            echo "  ✅ Bridge $bridge brought up (nmcli)"
        else
            # Try ip link as fallback
            if ip link set "$bridge" up 2>/dev/null; then
                echo "  ✅ Bridge $bridge brought up (ip link)"
            else
                echo "  ❌ Failed to bring up bridge $bridge"
                return 1
            fi
        fi
    fi
    
    # Check slave if specified
    if [[ -n "$slave" ]]; then
        # Check if slave interface exists
        if ! ip link show "$slave" &>/dev/null; then
            echo "  ⚠️  Slave interface $slave does not exist - skipping"
            return 0
        fi
        
        # Wait briefly for state to settle
        sleep 1
        
        # Check if slave is attached to bridge
        if bridge link show "$bridge" 2>/dev/null | grep -q "$slave"; then
            echo "  ✅ $slave is enslaved to $bridge"
            # Ensure slave is UP
            if ! ip link show "$slave" | grep -q "UP"; then
                echo "  ⚠️  $slave is enslaved but DOWN - bringing up..."
                ip link set "$slave" up 2>/dev/null
                if ip link show "$slave" | grep -q "UP"; then
                    echo "  ✅ $slave is now UP"
                fi
            fi
        else
            echo "  ⚠️  $slave NOT enslaved - attempting re-attachment..."
            
            # Re-attach slave to bridge
            if ip link set "$slave" master "$bridge" 2>/dev/null; then
                echo "  ✅ $slave re-enslaved to $bridge"
                
                # Bring slave up after attachment
                ip link set "$slave" up 2>/dev/null
                sleep 1
                if ip link show "$slave" | grep -q "UP"; then
                    echo "  ✅ $slave is UP"
                else
                    echo "  ⚠️  $slave is enslaved but DOWN"
                fi
            else
                echo "  ❌ Failed to re-enslave $slave to $bridge"
                return 1
            fi
        fi
    fi
    
    # Final verification
    sleep 1
    if ip link show "$bridge" | grep -q "UP"; then
        echo "  ✅ Bridge $bridge is operational"
        # Show final slave list
        local final_slaves=$(bridge link show "$bridge" 2>/dev/null | grep -oP '^[0-9]+: \K[^:]+' | tr '\n' ' ')
        echo "  ℹ️  Slaves: ${final_slaves:-none}"
        return 0
    else
        echo "  ❌ Bridge $bridge is still DOWN"
        return 1
    fi
}

#!/bin/bash
# repairs/dhcp - DHCP repair action with static IP awareness
# Usage: repair_dhcp [iface]

repair_dhcp() {
    local iface="${1:-${BRIDGE_IFACE:-$DEFAULT_IFACE}}"
    echo "🔧 Repairing IP on $iface..."
    
    # Detect if DHCP or static
    local ip_method="static"
    if command -v nmcli &>/dev/null; then
        if nmcli -t -f DHCP4 device show "$iface" 2>/dev/null | grep -q 'DHCP4.OPTION'; then
            ip_method="dhcp"
        fi
    else
        # Fallback: check if dhclient is running or if interface has DHCP lease
        if pgrep -f "dhclient.*$iface" &>/dev/null; then
            ip_method="dhcp"
        fi
    fi
    
    echo "  ℹ️  IP method: $ip_method"
    
    if [[ "$ip_method" == "dhcp" ]]; then
        if ! command -v dhclient &>/dev/null; then
            echo "  ❌ dhclient not installed"
            return 1
        fi
        echo "  📌 Renewing DHCP lease..."
        dhclient -r "$iface" 2>/dev/null
        if timeout 10 dhclient "$iface" 2>/dev/null; then
            # Verify IP was assigned
            local new_ip=$(ip -4 addr show "$iface" 2>/dev/null | grep -oP '(?<=inet\s)[0-9.]+' | head -1)
            if [[ -n "$new_ip" ]]; then
                echo "  ✅ DHCP lease renewed on $iface"
                echo "  ℹ️  New IP: $new_ip"
                return 0
            else
                echo "  ⚠️  DHCP renewed but no IP assigned"
                return 1
            fi
        else
            echo "  ❌ DHCP renewal failed (timeout or error)"
            return 1
        fi
    else
        # Static IP - restart the connection via nmcli
        echo "  📌 Restarting static IP connection..."
        if command -v nmcli &>/dev/null; then
            local conn_name=$(nmcli -t -f NAME,DEVICE connection show --active 2>/dev/null | grep ":${iface}$" | cut -d: -f1)
            if [[ -n "$conn_name" ]]; then
                nmcli con down "$conn_name" 2>/dev/null
                sleep 1
                if nmcli con up "$conn_name" 2>/dev/null; then
                    echo "  ✅ Static IP connection '$conn_name' restarted"
                    local current_ip=$(ip -4 addr show "$iface" 2>/dev/null | grep -oP '(?<=inet\s)[0-9.]+' | head -1)
                    [[ -n "$current_ip" ]] && echo "  ℹ️  IP: $current_ip"
                    return 0
                else
                    echo "  ❌ Failed to restart connection '$conn_name'"
                    echo "  📌 Attempting interface reset..."
                    ip link set "$iface" down 2>/dev/null
                    sleep 1
                    ip link set "$iface" up 2>/dev/null
                    if ip link show "$iface" | grep -q "UP"; then
                        echo "  ✅ Interface $iface reset"
                        return 0
                    fi
                fi
            else
                echo "  ⚠️  No active connection found for $iface"
                echo "  📌 Attempting interface reset..."
                ip link set "$iface" down 2>/dev/null
                sleep 1
                ip link set "$iface" up 2>/dev/null
                if ip link show "$iface" | grep -q "UP"; then
                    echo "  ✅ Interface $iface reset"
                    return 0
                fi
            fi
        else
            # No nmcli - use ip link as fallback
            echo "  📌 Resetting interface (ip link)..."
            ip link set "$iface" down 2>/dev/null
            sleep 1
            ip link set "$iface" up 2>/dev/null
            if ip link show "$iface" | grep -q "UP"; then
                echo "  ✅ Interface $iface reset"
                return 0
            fi
        fi
        echo "  ❌ Failed to restart static IP connection"
        return 1
    fi
}

#!/bin/bash
# repairs/dns - DNS repair action
# Usage: repair_dns

repair_dns() {
    echo "🔧 Repairing DNS..."
    local fixed=false
    
    # Flush DNS cache via resolvectl
    if command -v resolvectl &>/dev/null; then
        if resolvectl flush-caches 2>/dev/null; then
            echo "  ✅ DNS cache flushed (resolvectl)"
            fixed=true
        else
            echo "  ⚠️  resolvectl flush failed (may not be needed)"
        fi
    fi
    
    # Flush DNS cache via systemd-resolve (legacy)
    if command -v systemd-resolve &>/dev/null; then
        if systemd-resolve --flush-caches 2>/dev/null; then
            echo "  ✅ DNS cache flushed (systemd-resolve)"
            fixed=true
        else
            echo "  ⚠️  systemd-resolve flush failed"
        fi
    fi
    
    # Restart systemd-resolved service
    if systemctl restart systemd-resolved 2>/dev/null; then
        echo "  ✅ systemd-resolved restarted"
        fixed=true
    else
        echo "  ⚠️  systemd-resolved restart failed (may not be installed)"
    fi
    
    # Verify DNS is working after repair
    echo "  🔍 Verifying DNS resolution..."
    if nslookup google.com &>/dev/null || dig google.com +short &>/dev/null; then
        echo "  ✅ DNS resolution verified"
        fixed=true
        return 0
    else
        echo "  ❌ DNS still failing after repair"
        return 1
    fi
}

#!/bin/bash
# repairs/nm - NetworkManager repair action
# Usage: repair_nm

repair_nm() {
    echo "🔧 Repairing NetworkManager..."
    
    # Check current status
    if systemctl is-active NetworkManager &>/dev/null; then
        echo "  ℹ️  NetworkManager is currently running"
    else
        echo "  ⚠️  NetworkManager is not running"
    fi
    
    # Restart NetworkManager
    if systemctl restart NetworkManager 2>/dev/null; then
        echo "  ✅ NetworkManager restart initiated"
        
        # Wait for service to stabilize
        echo "  ⏳ Waiting for NetworkManager to stabilize..."
        local max_wait=10
        local wait_count=0
        
        while [[ $wait_count -lt $max_wait ]]; do
            if systemctl is-active NetworkManager &>/dev/null; then
                echo "  ✅ NetworkManager is running"
                break
            fi
            sleep 1
            wait_count=$((wait_count + 1))
        done
        
        if [[ $wait_count -eq $max_wait ]]; then
            echo "  ⚠️  NetworkManager did not stabilize within ${max_wait}s"
            echo "  ℹ️  Check status: systemctl status NetworkManager"
            return 1
        fi
        
        # Verify NM state via nmcli
        if command -v nmcli &>/dev/null; then
            sleep 2  # Give NM time to fully initialize
            local nm_state=$(nmcli -t -f GENERAL.STATE general status 2>/dev/null | cut -d: -f2)
            if [[ -n "$nm_state" ]]; then
                echo "  ℹ️  NetworkManager state: $nm_state"
                if [[ "$nm_state" == "connected" ]] || [[ "$nm_state" == "connecting" ]]; then
                    echo "  ✅ NetworkManager is connected"
                    return 0
                else
                    echo "  ⚠️  NetworkManager state is '$nm_state' - may need connection"
                    return 1
                fi
            fi
        fi
        
        return 0
    else
        echo "  ❌ Failed to restart NetworkManager"
        
        # Try fallback: kill and restart
        echo "  📌 Attempting fallback restart..."
        if pkill -f "NetworkManager" 2>/dev/null; then
            sleep 2
            if systemctl start NetworkManager 2>/dev/null; then
                echo "  ✅ NetworkManager restarted (fallback)"
                return 0
            fi
        fi
        
        echo "  ❌ All restart attempts failed"
        return 1
    fi
}

#!/bin/bash
# repairs/routing - routing repair action

repair_routing() {

    local gateway=$(ip route | awk '/default/ {print $3; exit}' 2>/dev/null)

                    echo "  ✅ NetworkManager is running"
                    
                    # Check if gateway appeared
                    local new_gateway=$(ip route | awk '/default/ {print $3; exit}' 2>/dev/null)
                    if [[ -n "$new_gateway" ]]; then
                        echo "  ✅ Default gateway restored: $new_gateway"
                        return 0
                    else
                        echo "  ⚠️  No default gateway after restart"
                        return 1
                    fi
                else
                    echo "  ❌ NetworkManager failed to start"
                    return 1
                fi
            else
                echo "  ❌ Failed to restart NetworkManager"
                return 1
            fi
        else
            echo "  ❌ systemctl not available - cannot restart NetworkManager"
            return 1

        # Verify gateway is reachable
        echo "  📌 Testing gateway reachability..."
        if ping -c 1 -W 2 "$gateway" &>/dev/null; then
            echo "  ✅ Gateway $gateway reachable - routing OK"
            return 0
        else
            echo "  ⚠️  Gateway unreachable - attempting repair..."
            
            # Try ARP refresh first
            echo "  📌 Refreshing ARP entry for gateway..."
            if arping -c 1 -I "$DEFAULT_IFACE" "$gateway" &>/dev/null 2>&1; then
                echo "  ✅ ARP entry refreshed"
                sleep 1
                
                # Retest gateway
                if ping -c 1 -W 2 "$gateway" &>/dev/null; then
                    echo "  ✅ Gateway now reachable"
                    return 0
                fi
            fi
            
            # If still unreachable, restart NM
            echo "  📌 Restarting NetworkManager to rebuild routes..."
            if command -v systemctl &>/dev/null; then
                if systemctl restart NetworkManager 2>/dev/null; then
                    echo "  ✅ NetworkManager restarted"
                    sleep 3
                    
                    # Check if gateway is now reachable
                    if ping -c 1 -W 2 "$gateway" &>/dev/null; then
                        echo "  ✅ Gateway now reachable"
                        return 0
                    else
                        echo "  ❌ Gateway still unreachable after restart"
                        return 1
                    fi
                else
                    echo "  ❌ Failed to restart NetworkManager"
                    return 1
                fi
            else
                echo "  ❌ systemctl not available - cannot restart NetworkManager"
                return 1
            fi
        fi
    fi
}

# ============================================================
# LAYER 1: PHYSICAL LAYER
# ============================================================

check_physical() {
    log "INFO" "=== LAYER 1: PHYSICAL ==="
    local all_pass=true
    
    # Check if interface exists
    if [[ ! -d "/sys/class/net/$DEFAULT_IFACE" ]]; then
        log_result "FAIL" "Interface $DEFAULT_IFACE does not exist"
        return 1
    fi
    
    # Check carrier
    if [[ -f "/sys/class/net/$DEFAULT_IFACE/carrier" ]]; then
        local carrier=$(cat "/sys/class/net/$DEFAULT_IFACE/carrier")
        if [[ "$carrier" == "1" ]]; then
            log_result "PASS" "Carrier: present"
        else
            log_result "FAIL" "Carrier: LOST - Check cable/WiFi"
            all_pass=false
        fi
    fi
    
    # Check operstate
    if [[ -f "/sys/class/net/$DEFAULT_IFACE/operstate" ]]; then
        local state=$(cat "/sys/class/net/$DEFAULT_IFACE/operstate")
        if [[ "$state" == "up" ]]; then
            log_result "PASS" "Interface state: $state"
        else
            log_result "FAIL" "Interface state: $state (should be up)"
            all_pass=false
        fi
    fi
    
    # Check speed and duplex (ethernet only) - with timeout to prevent hangs
    if command -v ethtool &>/dev/null && [[ -e "/sys/class/net/$DEFAULT_IFACE/device" ]]; then
        local speed=$(timeout 2 ethtool "$DEFAULT_IFACE" 2>/dev/null | grep -i "Speed:" | awk '{print $2}')
        local duplex=$(timeout 2 ethtool "$DEFAULT_IFACE" 2>/dev/null | grep -i "Duplex:" | awk '{print $2}')
        if [[ -n "$speed" ]]; then
            log_result "INFO" "Speed: $speed, Duplex: $duplex"
        fi
    fi
    
    if [[ "$all_pass" == "true" ]]; then
        return 0
    else
        return 1
    fi
}

# ============================================================
# LAYER 2: LINK LAYER (Bridge, Bond, VLAN)
# ============================================================

check_link_layer() {
    log "INFO" "=== LAYER 2: LINK LAYER ==="
    local all_pass=true
    
    # Check if interface is UP
    if ip link show "$DEFAULT_IFACE" | grep -q "UP"; then
        log_result "PASS" "Interface $DEFAULT_IFACE is UP"
    else
        log_result "FAIL" "Interface $DEFAULT_IFACE is DOWN"
        all_pass=false
    fi
    
    # Check bridge if present
    if [[ -n "$BRIDGE_IFACE" ]]; then
        if ip link show "$BRIDGE_IFACE" | grep -q "UP"; then
            log_result "PASS" "Bridge $BRIDGE_IFACE is UP"
        else
            log_result "FAIL" "Bridge $BRIDGE_IFACE is DOWN"
            all_pass=false
        fi
        
        # Check if interface is enslaved to bridge
        if bridge link show "$BRIDGE_IFACE" 2>/dev/null | grep -q "$DEFAULT_IFACE"; then
            log_result "PASS" "$DEFAULT_IFACE enslaved to $BRIDGE_IFACE"
        else
            log_result "WARN" "$DEFAULT_IFACE NOT enslaved to $BRIDGE_IFACE"
            all_pass=false
        fi
    fi
    
    # Check for vnet interfaces (KVM)
    if [[ "$HAS_LIBVIRT" == "true" ]]; then
        local vnets=$(ip link show | grep -oP 'vnet\d+' | sort -u)
        if [[ -n "$vnets" ]]; then
            for vnet in $vnets; do
                if bridge link show 2>/dev/null | grep -q "$vnet"; then
                    log_result "PASS" "$vnet is bridged"
                else
                    log_result "WARN" "$vnet NOT bridged"
                    all_pass=false
                fi
            done
        fi
    fi
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

# LAYER 3: IP LAYER
# ============================================================

check_ip_layer() {
    log "INFO" "=== LAYER 3: IP ==="
    local all_pass=true
    
    # Check IPv4 address
    local iface_to_check="${BRIDGE_IFACE:-$DEFAULT_IFACE}"
    local ipv4=$(ip -4 addr show "$iface_to_check" 2>/dev/null | grep -oP '(?<=inet\s)\d+(\.\d+){3}/\d+' | head -1)
    
    if [[ -n "$ipv4" ]]; then
        log_result "PASS" "IPv4: $ipv4"
        
        # Check for duplicate IP - FIXED: compare MAC addresses
        local ip_only=$(echo "$ipv4" | cut -d/ -f1)
        local my_mac=$(ip link show "$iface_to_check" 2>/dev/null | grep -oP 'link/ether \K[^ ]+')
        if [[ -n "$my_mac" ]]; then
            local arp_output=$(arping -c 2 -I "$iface_to_check" "$ip_only" 2>/dev/null)
            local arp_mac=$(echo "$arp_output" | grep -oP 'from \K[^ ]+' | head -1)
            if [[ -n "$arp_mac" ]] && [[ "$arp_mac" != "$my_mac" ]]; then
                log_result "FAIL" "Duplicate IP detected! Own MAC: $my_mac, Duplicate MAC: $arp_mac"
                all_pass=false
            fi
        fi
    else
        log_result "FAIL" "No IPv4 address on $iface_to_check"
        all_pass=false
    fi
    
    # Check IPv6
    local ipv6=$(ip -6 addr show "$iface_to_check" 2>/dev/null | grep -oP '(?<=inet6\s)[a-f0-9:]+' | grep -v "^::1" | head -1)
    if [[ -n "$ipv6" ]]; then
        log_result "INFO" "IPv6: $ipv6"
    else
        log_result "INFO" "No IPv6 address (may be disabled)"
    fi
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

# ============================================================
# LAYER 5: GATEWAY
# ============================================================

check_gateway() {
    log "INFO" "=== LAYER 5: GATEWAY ==="
    
    if [[ -z "$DEFAULT_GATEWAY" ]]; then
        log_result "FAIL" "No default gateway - skipping gateway checks"
        return 1
    fi
    
    local all_pass=true
    
    # Check ARP entry
    if ip neigh show "$DEFAULT_GATEWAY" | grep -q "REACHABLE"; then
        log_result "PASS" "Gateway ARP: REACHABLE"
    elif ip neigh show "$DEFAULT_GATEWAY" | grep -q "STALE"; then
        log_result "WARN" "Gateway ARP: STALE - may need refresh"
        all_pass=false
    else
        log_result "WARN" "No ARP entry for gateway"
        all_pass=false
    fi
    
    # Ping gateway with short timeout
    if ping -c 1 -W "$PING_TIMEOUT" "$DEFAULT_GATEWAY" &>/dev/null; then
        log_result "PASS" "Gateway reachable (ICMP)"
    else
        # Try ARP ping
        if arping -c 1 -W 1 "$DEFAULT_GATEWAY" &>/dev/null; then
            log_result "WARN" "Gateway responds to ARP but not ICMP"
        else
            log_result "FAIL" "Gateway unreachable"
            all_pass=false
        fi
    fi
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

# ============================================================
# LAYER 6: INTERNET REACHABILITY
# ============================================================

check_internet() {
    log "INFO" "=== LAYER 6: INTERNET REACHABILITY ==="
    local reachable_count=0
    local total_targets=${#IP_TARGETS[@]}
    
    for target in "${IP_TARGETS[@]}"; do
        if ping -c 1 -W "$PING_TIMEOUT" "$target" &>/dev/null; then
            log_result "PASS" "$target reachable (ICMP)"
            ((reachable_count++))
        else
            # Try TCP connect as fallback
            if command -v nc &>/dev/null && nc -zv -w 3 "$target" 443 &>/dev/null 2>&1; then
                log_result "PASS" "$target reachable (TCP/443)"
                ((reachable_count++))
            else
                log_result "WARN" "$target unreachable"
            fi
        fi
    done
    
    # Require quorum (at least 2 out of 3)
    if [[ "$reachable_count" -ge 2 ]]; then
        log_result "PASS" "Internet reachable ($reachable_count/$total_targets)"
        return 0
    else
        log_result "FAIL" "Internet unreachable ($reachable_count/$total_targets)"
        return 1
    fi
}

# LAYER 7: DNS
# ============================================================

check_dns() {
    log "INFO" "=== LAYER 7: DNS ==="
    local all_pass=true
    
    # Check resolv.conf
    if [[ -f /etc/resolv.conf ]]; then
        local nameservers=$(grep "^nameserver" /etc/resolv.conf | wc -l)
        if [[ "$nameservers" -gt 0 ]]; then
            log_result "PASS" "DNS servers configured ($nameservers)"
            grep "^nameserver" /etc/resolv.conf | while read -r line; do
                log_result "INFO" "  $line"
            done
        else
            log_result "FAIL" "No nameservers in /etc/resolv.conf"
            all_pass=false
        fi
    fi
    
    # Test DNS resolution with multiple servers
    local dns_success=0
    for domain in "${DNS_TARGETS[@]}"; do
        if nslookup "$domain" &>/dev/null; then
            log_result "PASS" "$domain resolves"
            ((dns_success++))
        else
            # Try with specific DNS server
            if dig "@1.1.1.1" "$domain" +timeout=2 &>/dev/null; then
                log_result "PASS" "$domain resolves (via 1.1.1.1)"
                ((dns_success++))
            else
                log_result "WARN" "$domain resolution failed"
            fi
        fi
    done
    
    # Check for DNS poisoning
    if command -v dig &>/dev/null; then
        local ip1=$(dig "@1.1.1.1" google.com +short 2>/dev/null | head -1)
        local ip2=$(dig "@8.8.8.8" google.com +short 2>/dev/null | head -1)
        if [[ -n "$ip1" ]] && [[ -n "$ip2" ]] && [[ "$ip1" != "$ip2" ]]; then
            log_result "WARN" "DNS poisoning suspected - different answers from different resolvers"
            log_result "INFO" "  1.1.1.1: $ip1"
            log_result "INFO" "  8.8.8.8: $ip2"
        fi
    fi
    
    # Check systemd-resolved status
    if command -v resolvectl &>/dev/null; then
        local dns_state=$(resolvectl status 2>/dev/null | grep -i "DNS Servers" | head -1)
        if [[ -n "$dns_state" ]]; then
            log_result "INFO" "systemd-resolved: $dns_state"
        fi
    fi
    
    if [[ "$dns_success" -ge 2 ]]; then
        return 0
    else
        return 1
    fi
}

# LAYER 8: HTTPS / TLS / ACTUAL BROWSING
# ============================================================

check_https() {
    log "INFO" "=== LAYER 8: HTTPS & BROWSING ==="
    local all_pass=true
    local https_success=0
    
    # Test captive portal detection
    if command -v curl &>/dev/null; then
        local portal_response=$(curl -s -o /dev/null -w "%{http_code}" --connect-timeout 5 "$CAPTIVE_PORTAL_TEST" 2>/dev/null)
        if [[ "$portal_response" == "204" ]]; then
            log_result "PASS" "No captive portal detected"
        elif [[ -n "$portal_response" ]]; then
            log_result "WARN" "Captive portal detected (HTTP $portal_response)"
            all_pass=false
        fi
    fi
    
    # Test HTTPS targets
    for target in "${HTTPS_TARGETS[@]}"; do
        if command -v curl &>/dev/null; then
            local http_code=$(curl -s -o /dev/null -w "%{http_code}" --connect-timeout "$CURL_TIMEOUT" --max-time 10 "$target" 2>/dev/null)
            if [[ "$http_code" =~ ^[23][0-9][0-9]$ ]]; then
                log_result "PASS" "HTTPS: $target (HTTP $http_code)"
                ((https_success++))
            else
                log_result "WARN" "HTTPS: $target failed (HTTP $http_code)"
                all_pass=false
            fi
        elif command -v wget &>/dev/null; then
            if wget --spider --timeout=5 "$target" 2>/dev/null; then
                log_result "PASS" "HTTPS: $target reachable"
                ((https_success++))
            else
                log_result "WARN" "HTTPS: $target failed"
                all_pass=false
            fi
        else
            log_result "WARN" "curl/wget not installed - skipping HTTPS tests"
            all_pass=false
            break
        fi
    done
    
    # Check for MTU issues
    if [[ "$https_success" -lt 2 ]] && [[ -n "$DEFAULT_GATEWAY" ]]; then
        log_result "INFO" "Testing MTU..."
        if ping -M do -s 1472 -c 1 "$DEFAULT_GATEWAY" &>/dev/null; then
            log_result "PASS" "MTU: 1500 works"
        else
            log_result "WARN" "MTU issues detected - try reducing MTU to 1400"
            all_pass=false
        fi
    fi
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

check_networkmanager() {
    log "INFO" "=== LAYER 9: NETWORKMANAGER ==="
    local all_pass=true
    
    if ! command -v nmcli &>/dev/null; then
        log_result "INFO" "NetworkManager not installed (using other network stack)"
        return 0
    fi
    
    # Check if NM is running
    if systemctl is-active NetworkManager &>/dev/null; then
        log_result "PASS" "NetworkManager is running"
    else
        log_result "FAIL" "NetworkManager is NOT running"
        return 1
    fi
    
    # Check NM state
    local nm_state=$(nmcli -t -f GENERAL.STATE general status 2>/dev/null | cut -d: -f2)
    if [[ -n "$nm_state" ]]; then
        log_result "INFO" "NetworkManager state: $nm_state"
        if [[ "$nm_state" == "connected" ]] || [[ "$nm_state" == "connecting" ]]; then
            log_result "PASS" "NetworkManager has connection"
        else
            log_result "WARN" "NetworkManager state is '$nm_state'"
            all_pass=false
        fi
    fi
    
    # Check device status
    nmcli device status 2>/dev/null | while read -r line; do
        if [[ "$line" =~ ^[^ ]+[[:space:]]+[^ ]+[[:space:]]+connected ]]; then
            log_result "PASS" "$line"
        fi
    done
    
    # Check for stale connections
    local stale_connections=$(nmcli connection show --active 2>/dev/null | grep -c " (stale)")
    if [[ "$stale_connections" -gt 0 ]]; then
        log_result "WARN" "Stale connections detected ($stale_connections)"
        all_pass=false
    fi
    
    [[ "$all_pass" == "true" ]] && return 0 || return 1
}

        log_result "INFO" "Kubernetes: $nodes/$total nodes ready"
    else
        log_result "WARN" "Kubernetes API not accessible"
        return 1
    fi
    
    return 0
}

# ============================================================
# REPAIR ENGINE - Layer-aware, static IP aware
# ============================================================

repair_network() {
    log "INFO" "=== REPAIR ENGINE ==="
    log "INFO" "Starting targeted repairs based on diagnostic results..."
    local repair_success_count=0
    local repair_attempt_count=0
    local iface_to_repair="${BRIDGE_IFACE:-$DEFAULT_IFACE}"
    local ip_method=$(detect_ip_method "$iface_to_repair")
    
    log "INFO" "IP method detected: $ip_method on $iface_to_repair"
    
    # Stage 1: DNS repairs (only if DNS failed)
    if [[ ${LAYER_STATUS[dns]:-0} -ne 0 ]]; then
        log "INFO" "Stage 1: DNS repairs (DNS layer failed)"
        ((repair_attempt_count++))
        local dns_fixed=false
        
        if command -v resolvectl &>/dev/null; then
            if resolvectl flush-caches 2>/dev/null; then
                log_result "PASS" "DNS cache flushed (resolvectl)"
                dns_fixed=true
            fi
        fi
        if command -v systemd-resolve &>/dev/null; then
            if systemd-resolve --flush-caches 2>/dev/null; then
                log_result "PASS" "DNS cache flushed (systemd-resolve)"
                dns_fixed=true
            fi
        fi
        if command -v nscd &>/dev/null; then
            if systemctl restart nscd 2>/dev/null; then
                log_result "PASS" "nscd restarted"
                dns_fixed=true
            fi
        fi
        if systemctl restart systemd-resolved 2>/dev/null; then
            log_result "PASS" "systemd-resolved restarted"
            dns_fixed=true
        fi
        
        if [[ "$dns_fixed" == "true" ]]; then
            ((repair_success_count++))
        else
            log_result "WARN" "DNS repairs had no effect"
        fi
    else
        log "INFO" "Stage 1: DNS OK - skipping DNS repairs"
    fi
    
    # Stage 2: ARP cache (only if gateway or routing failed)
    if [[ ${LAYER_STATUS[gateway]:-0} -ne 0 ]] || [[ ${LAYER_STATUS[routing]:-0} -ne 0 ]]; then
        log "INFO" "Stage 2: Flushing ARP cache (gateway/routing layer failed)"
        ((repair_attempt_count++))
        if ip neigh flush all 2>/dev/null; then
            log_result "PASS" "ARP cache flushed"
            ((repair_success_count++))
        else
            log_result "WARN" "Failed to flush ARP cache"
        fi
    else
        log "INFO" "Stage 2: Gateway/Routing OK - skipping ARP flush"
    fi
    
    # Stage 3: Interface reconnect (if physical or link layer failed)
    if [[ ${LAYER_STATUS[physical]:-0} -ne 0 ]] || [[ ${LAYER_STATUS[link]:-0} -ne 0 ]]; then
        log "INFO" "Stage 3: Reconnecting interface (physical/link layer failed)"
        ((repair_attempt_count++))
        if command -v nmcli &>/dev/null; then
            if nmcli device status 2>/dev/null | grep -q "$DEFAULT_IFACE"; then
                nmcli device disconnect "$DEFAULT_IFACE" 2>/dev/null
                sleep 1
                if nmcli device connect "$DEFAULT_IFACE" 2>/dev/null; then
                    log_result "PASS" "Interface $DEFAULT_IFACE reconnected"
                    ((repair_success_count++))
                else
                    log_result "WARN" "Failed to reconnect $DEFAULT_IFACE"
                fi
                sleep 2
            fi
        fi
    else
        log "INFO" "Stage 3: Physical/Link OK - skipping interface reconnect"
    fi
    
    # Stage 4: IP/DHCP repair (only if IP layer failed)
    if [[ ${LAYER_STATUS[ip]:-0} -ne 0 ]]; then
        log "INFO" "Stage 4: IP repair (IP layer failed)"
        ((repair_attempt_count++))
        
        if [[ "$ip_method" == "dhcp" ]]; then
            log "INFO" "Renewing DHCP lease on $iface_to_repair"
            if command -v dhclient &>/dev/null; then
                dhclient -r "$iface_to_repair" 2>/dev/null
                if dhclient "$iface_to_repair" 2>/dev/null; then
                    log_result "PASS" "DHCP lease renewed on $iface_to_repair"
                    ((repair_success_count++))
                else
                    log_result "WARN" "Failed to renew DHCP lease"
                fi
            fi
        else
            log "INFO" "Static IP detected - restarting connection instead"
            if command -v nmcli &>/dev/null; then
                local conn_name=$(nmcli -t -f NAME,DEVICE connection show --active 2>/dev/null | grep ":${iface_to_repair}$" | cut -d: -f1)
                if [[ -n "$conn_name" ]]; then
                    nmcli con down "$conn_name" 2>/dev/null
                    sleep 1
                    if nmcli con up "$conn_name" 2>/dev/null; then
                        log_result "PASS" "Static IP connection '$conn_name' restarted"
                        ((repair_success_count++))
                    else
                        log_result "WARN" "Failed to restart connection '$conn_name'"
                    fi
                else
                    log_result "WARN" "Could not find connection name for $iface_to_repair"
                fi
            fi
        fi
    else
        log "INFO" "Stage 4: IP layer OK - skipping IP repair"
    fi
    
    # Stage 5: NetworkManager restart (if NM layer failed)
    if [[ ${LAYER_STATUS[nm]:-0} -ne 0 ]]; then
        log "INFO" "Stage 5: Restarting NetworkManager (NM layer failed)"
        ((repair_attempt_count++))
        if systemctl restart NetworkManager 2>/dev/null; then
            log_result "PASS" "NetworkManager restarted"
            ((repair_success_count++))
            sleep 3
        else
            log_result "WARN" "Failed to restart NetworkManager"
        fi
    else
        log "INFO" "Stage 5: NetworkManager OK - skipping NM restart"
    fi
    
    # Stage 6: Resolver restart (if DNS failed and not already done)
    if [[ ${LAYER_STATUS[dns]:-0} -ne 0 ]]; then
        log "INFO" "Stage 6: Restarting resolver (DNS layer failed)"
        ((repair_attempt_count++))
        if systemctl restart systemd-resolved 2>/dev/null; then
            log_result "PASS" "systemd-resolved restarted"
            ((repair_success_count++))
        fi
    fi
    
    # Stage 7: Bridge recovery (if bridge issues detected)
    if [[ -n "$BRIDGE_IFACE" ]] && [[ ${LAYER_STATUS[link]:-0} -ne 0 ]]; then
        log "INFO" "Stage 7: Bridge recovery (link layer failed)"
        ((repair_attempt_count++))
        if nmcli con up "$BRIDGE_IFACE" 2>/dev/null; then
            log_result "PASS" "Bridge $BRIDGE_IFACE brought up"
            ((repair_success_count++))
            
            # Re-enslave interface if needed
            if ! bridge link show "$BRIDGE_IFACE" 2>/dev/null | grep -q "$DEFAULT_IFACE"; then
                log "INFO" "Re-enslaving $DEFAULT_IFACE to $BRIDGE_IFACE"
                if ip link set "$DEFAULT_IFACE" master "$BRIDGE_IFACE" 2>/dev/null; then
                    log_result "PASS" "$DEFAULT_IFACE re-enslaved to $BRIDGE_IFACE"
                else
                    log_result "WARN" "Failed to re-enslave $DEFAULT_IFACE"
                fi
            fi
        else
            log_result "WARN" "Failed to bring up bridge $BRIDGE_IFACE"
        fi
    fi
    
    # Stage 8: DNS fallback (last resort, only if DNS still broken)
    if [[ ${LAYER_STATUS[dns]:-0} -ne 0 ]]; then
        log "INFO" "Stage 8: DNS fallback configuration"
        if [[ ! -f /etc/resolv.conf ]]; then
            echo "nameserver 1.1.1.1" > /etc/resolv.conf
            echo "nameserver 8.8.8.8" >> /etc/resolv.conf
            log_result "PASS" "Fallback DNS configured (new resolv.conf)"
            ((repair_success_count++))
            ((repair_attempt_count++))
        elif [[ -L /etc/resolv.conf ]]; then
            log_result "INFO" "resolv.conf is a symlink (managed by resolver) - not modifying"
        elif ! grep -q "^nameserver" /etc/resolv.conf; then
            echo "nameserver 1.1.1.1" >> /etc/resolv.conf
            echo "nameserver 8.8.8.8" >> /etc/resolv.conf
            log_result "PASS" "Fallback DNS added to resolv.conf"
            ((repair_success_count++))
            ((repair_attempt_count++))
        else
            log_result "INFO" "resolv.conf already has nameservers - skipping"
        fi
    fi
    
    # Report results
    if [[ "$repair_success_count" -gt 0 ]]; then
        log_result "PASS" "Repairs completed: $repair_success_count/$repair_attempt_count succeeded"
        return 0
    elif [[ "$repair_attempt_count" -eq 0 ]]; then
        log_result "INFO" "No repairs needed - all layers healthy"
        return 0
    else
        log_result "FAIL" "No repairs succeeded (0/$repair_attempt_count)"
        return 1
    fi
}

        echo "Internet: ❌ Unreachable"

    if nslookup google.com &>/dev/null 2>&1; then
        echo "DNS:      ✅ Working"
    else
        echo "DNS:      ❌ Failed"
    fi
    
    if systemctl is-active NetworkManager &>/dev/null; then
        local nm_state=$(nmcli -t -f GENERAL.STATE general status 2>/dev/null | cut -d: -f2)
        echo "NM:       ✅ Running (${nm_state:-unknown})"
    else
        echo "NM:       ❌ Stopped"
    fi
    
    if [[ "$HAS_LIBVIRT" == "true" ]]; then
        if systemctl is-active libvirtd &>/dev/null; then
            local vm_count=$(virsh list --state-running --name 2>/dev/null | wc -l)
            echo "VMs:      ✅ $vm_count running"
        else
            echo "VMs:      ❌ libvirtd stopped"
        fi
    fi
    
    echo "=========================================="
}

cmd_diagnose() {
    log "INFO" "=== NETWORK DIAGNOSTIC STARTED ==="
    local TIMESTAMP=$(generate_timestamp)
    local results_file="$LOG_DIR/diagnostic-${TIMESTAMP}.log"
    
    {
        check_physical
        check_link_layer

    } | tee -a "$results_file"

}

cmd_repair() {
    log "INFO" "=== NETWORK RECOVERY STARTED ==="
    local TIMESTAMP=$(generate_timestamp)
    
    # Take snapshot before any changes
    local snapshot_file=$(save_snapshot)
    log "INFO" "Pre-repair snapshot: $snapshot_file"
    
    # Run diagnostic and capture layer status
    local results_file="$LOG_DIR/recovery-${TIMESTAMP}.log"
    local diag_output=$(mktemp)
    trap 'rm -f "$diag_output"' EXIT
    
    # Initialize layer status array
    declare -gA LAYER_STATUS
    
    {
        check_physical; LAYER_STATUS[physical]=$?
        check_link_layer; LAYER_STATUS[link]=$?
        check_ip_layer; LAYER_STATUS[ip]=$?
        check_routing; LAYER_STATUS[routing]=$?
        check_gateway; LAYER_STATUS[gateway]=$?
        check_internet; LAYER_STATUS[internet]=$?
        check_dns; LAYER_STATUS[dns]=$?
        check_https; LAYER_STATUS[https]=$?
        check_networkmanager; LAYER_STATUS[nm]=$?
        check_virtualization; LAYER_STATUS[virt]=$?
        check_kubernetes; LAYER_STATUS[k8s]=$?
    } | tee -a "$diag_output"
    
    echo ""
    echo "=========================================="
    echo "    ATTEMPTING REPAIRS"
    echo "=========================================="
    
    repair_network
    local repair_result=$?
    
    echo ""
    echo "=========================================="
    echo "    VERIFYING REPAIRS"
    echo "=========================================="
    sleep 3
    
    # Re-run key tests
    echo -e "\n=== POST-REPAIR VERIFICATION ==="
    check_internet
    check_dns
    check_https
    
    cat "$diag_output" | tee -a "$results_file"
    generate_report "$results_file"
    
    return $repair_result
}

}

cmd_watch() {
    echo "🔍 Monitoring network health (Ctrl+C to stop)"
    echo ""
    while true; do
        clear
        echo "=== NETWORK HEALTH MONITOR ==="
        echo "Time: $(date)"
        echo ""
        
        # IP connectivity
        if ping -c 1 -W 1 8.8.8.8 &>/dev/null; then
            echo "  ✅ Internet: OK"
        else
            echo "  ❌ Internet: DOWN"
        fi
        
        # DNS
        if nslookup google.com &>/dev/null; then
            echo "  ✅ DNS: OK"
        else
            echo "  ❌ DNS: FAILED"
        fi
        
        # Gateway
        if [[ -n "$DEFAULT_GATEWAY" ]] && ping -c 1 -W 1 "$DEFAULT_GATEWAY" &>/dev/null; then
            echo "  ✅ Gateway: $DEFAULT_GATEWAY (reachable)"
        else
            echo "  ❌ Gateway: ${DEFAULT_GATEWAY:-'None'} (unreachable)"
        fi
        
        # Interface
        if ip link show "$DEFAULT_IFACE" | grep -q "UP"; then
            echo "  ✅ Interface: $DEFAULT_IFACE (UP)"
        else
            echo "  ❌ Interface: $DEFAULT_IFACE (DOWN)"
        fi
        
        echo ""
        echo "Press Ctrl+C to stop"
        sleep 2
    done
}

            ;;

            cmd_snapshot

            cmd_watch

        help|--help|-h)
            show_usage
            ;;

#!/bin/bash
# tests/test-offline.sh - Simulate offline conditions and test recovery
# Usage: ./test-offline.sh [--diagnose-only] [--interface IFACE]

set -euo pipefail

# Parse arguments
DIAGNOSE_ONLY=false
TEST_IFACE=""

while [[ $# -gt 0 ]]; do
    case $1 in
        --diagnose-only|-d)
            DIAGNOSE_ONLY=true
            shift
            ;;
        --interface|-i)
            TEST_IFACE="$2"
            shift 2
            ;;
        --help|-h)
            echo "Usage: $0 [--diagnose-only] [--interface IFACE]"
            echo "  --diagnose-only  Run diagnose only (no repairs)"
            echo "  --interface      Specify interface to test (default: auto-detect)"
            echo "  --help           Show this help"
            exit 0
            ;;
        *)
            echo "Unknown option: $1"
            echo "Use --help for usage"
            exit 1
            ;;
    esac
done

echo "=============================================="
echo "  NETWORK OFFLINE TEST"
echo "=============================================="
echo ""

# Detect interface
if [[ -z "$TEST_IFACE" ]]; then
    IFACE=$(ip route | awk '/default/ {print $5; exit}' 2>/dev/null || echo "wlan0")
else
    IFACE="$TEST_IFACE"
fi

echo "📌 Interface: $IFACE"

# Determine interface type
if [[ -d "/sys/class/net/$IFACE/wireless" ]] || [[ -d "/sys/class/net/$IFACE/phy80211" ]]; then
    IFACE_TYPE="wireless"
    echo "📌 Type: WiFi"
elif [[ -d "/sys/class/net/$IFACE/bridge" ]]; then
    IFACE_TYPE="bridge"
    echo "📌 Type: Bridge"
else
    IFACE_TYPE="ethernet"
    echo "📌 Type: Ethernet"
fi

# Check if interface exists
if [[ ! -d "/sys/class/net/$IFACE" ]]; then
    echo "❌ Interface $IFACE does not exist"
    exit 1
fi

# Save current state
echo ""
echo "📌 Current state:"
if ip link show "$IFACE" | grep -q "UP"; then
    echo "  ✅ Interface is UP"
else
    echo "  ⚠️  Interface is DOWN"
fi

IP_ADDR=$(ip -4 addr show "$IFACE" 2>/dev/null | grep -oP '(?<=inet\s)[0-9.]+' | head -1)
echo "  IP: ${IP_ADDR:-None}"

GATEWAY=$(ip route | awk '/default/ {print $3; exit}' 2>/dev/null || echo "None")
echo "  Gateway: $GATEWAY"

# Save connection name if using NetworkManager
CONN_NAME=""
if command -v nmcli &>/dev/null; then
    CONN_NAME=$(nmcli -t -f NAME,DEVICE connection show --active 2>/dev/null | grep ":${IFACE}$" | cut -d: -f1)
    if [[ -n "$CONN_NAME" ]]; then
        echo "  Connection: $CONN_NAME"
    fi
fi

# Test connectivity before disabling
echo ""
echo "📌 Testing connectivity before disabling..."
if ping -c 1 -W 2 8.8.8.8 &>/dev/null; then
    echo "  ✅ Internet reachable"
else
    echo "  ⚠️  Internet not reachable (test may be less useful)"
fi

echo ""
echo "=============================================="
echo "  SIMULATING OFFLINE CONDITIONS"
echo "=============================================="
echo ""

# Disable interface
echo "📌 Disabling $IFACE..."
if [[ "$IFACE_TYPE" == "wireless" ]]; then
    # For WiFi, use nmcli if available
    if command -v nmcli &>/dev/null; then
        sudo nmcli radio wifi off 2>/dev/null && echo "  ✅ WiFi disabled (nmcli)"
    else
        sudo ip link set "$IFACE" down 2>/dev/null && echo "  ✅ Interface down (ip link)"
    fi
else
    sudo ip link set "$IFACE" down 2>/dev/null && echo "  ✅ Interface down (ip link)"
fi

# Verify it's down
sleep 2
if ip link show "$IFACE" | grep -q "UP"; then
    echo "  ⚠️  Interface still UP - forced down may have failed"
fi

echo ""
echo "📌 Verifying network is down..."
if ping -c 1 -W 2 8.8.8.8 &>/dev/null; then
    echo "  ⚠️  Internet still reachable (test may not work)"
else
    echo "  ✅ Internet unreachable (as expected)"
fi

echo ""
echo "=============================================="
echo "  RUNNING NETWORK RECOVERY"
echo "=============================================="
echo ""

if [[ "$DIAGNOSE_ONLY" == "true" ]]; then
    echo "📌 Running diagnose only (no repairs)..."
    sudo /usr/local/bin/network-recover diagnose
else
    echo "📌 Running repair..."
    sudo /usr/local/bin/network-recover repair
fi

echo ""
echo "=============================================="
echo "  RESTORING NETWORK"
echo "=============================================="
echo ""

# Re-enable interface
echo "📌 Re-enabling $IFACE..."
if [[ "$IFACE_TYPE" == "wireless" ]]; then
    if command -v nmcli &>/dev/null; then
        sudo nmcli radio wifi on 2>/dev/null && echo "  ✅ WiFi enabled (nmcli)"
    else
        sudo ip link set "$IFACE" up 2>/dev/null && echo "  ✅ Interface up (ip link)"
    fi
else
    sudo ip link set "$IFACE" up 2>/dev/null && echo "  ✅ Interface up (ip link)"
fi

# Wait for interface to come up
echo "  ⏳ Waiting for interface to stabilize..."
sleep 3

# Re-activate connection
if [[ -n "$CONN_NAME" ]] && command -v nmcli &>/dev/null; then
    echo "📌 Re-activating connection '$CONN_NAME'..."
    sudo nmcli con down "$CONN_NAME" 2>/dev/null || true
    sleep 1
    if sudo nmcli con up "$CONN_NAME" 2>/dev/null; then
        echo "  ✅ Connection re-activated"
    else
        echo "  ⚠️  Failed to re-activate connection"
    fi
fi

# Wait for network to stabilize (longer for bridge setups with VMs)
if [[ "$IFACE_TYPE" == "bridge" ]]; then
    echo "  ⏳ Bridge detected - waiting longer for VM networking..."
    sleep 10
else
    echo "  ⏳ Waiting for network to stabilize..."
    sleep 5
fi

echo ""
echo "=============================================="
echo "  TEST RESULTS"
echo "=============================================="
echo ""

# Verify final state
echo "📌 Final state:"
if ip link show "$IFACE" | grep -q "UP"; then
    echo "  ✅ Interface is UP"
else
    echo "  ❌ Interface is DOWN"
fi

FINAL_IP=$(ip -4 addr show "$IFACE" 2>/dev/null | grep -oP '(?<=inet\s)[0-9.]+' | head -1)
if [[ -n "$FINAL_IP" ]]; then
    echo "  ✅ IP: $FINAL_IP"
else
    echo "  ❌ No IP assigned"
fi

if ping -c 1 -W 2 8.8.8.8 &>/dev/null; then
    echo "  ✅ Internet reachable"
    INTERNET_OK=true
else
    echo "  ❌ Internet unreachable"
    INTERNET_OK=false
fi

# DNS check with fallback
if command -v nslookup &>/dev/null; then
    if nslookup google.com &>/dev/null 2>&1; then
        echo "  ✅ DNS working"
    else
        echo "  ❌ DNS not working"
    fi
elif command -v dig &>/dev/null; then
    if dig google.com +short &>/dev/null 2>&1; then
        echo "  ✅ DNS working"
    else
        echo "  ❌ DNS not working"
    fi
else
    echo "  ⚠️  nslookup/dig not available - skipping DNS test"
fi

echo ""
echo "=============================================="
echo "  TEST COMPLETE"
echo "=============================================="

if [[ "$INTERNET_OK" == "true" ]]; then
    echo "✅ Network recovery test PASSED"
    echo ""
    echo "  Run 'sudo network-recover status' to verify connectivity"
    exit 0
else
    echo "❌ Network recovery test FAILED"
    echo ""
    echo "  Manual recovery options:"
    echo "    sudo network-recover diagnose"
    echo "    sudo network-recover repair"
    echo "    sudo network-recover status"
    exit 1
fi
